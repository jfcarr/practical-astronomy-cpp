23

25 double p0 = 130.143076320618;
26 double n0 = 291.682546643194;
27 double i = 5.145396;
28

30 lctHour, lctMin, lctSec, daylightSaving, zoneCorrectionHours,
31 localDateDay, localDateMonth, localDateYear);

33 lctHour, lctMin, lctSec, daylightSaving, zoneCorrectionHours,
34 localDateDay, localDateMonth, localDateYear);

36 lctHour, lctMin, lctSec, daylightSaving, zoneCorrectionHours,
37 localDateDay, localDateMonth, localDateYear);
38
39 double utHours = LocalCivilTimeToUniversalTime(
40 lctHour, lctMin, lctSec, daylightSaving, zoneCorrectionHours,
41 localDateDay, localDateMonth, localDateYear);
42 double dDays = CivilDateToJulianDate(gdateDay, gdateMonth, gdateYear) -
43 CivilDateToJulianDate(0.0, 1, 2010) + utHours / 24;
44 double sunLongDeg =
45 SunLong(lctHour, lctMin, lctSec, daylightSaving, zoneCorrectionHours,
46 localDateDay, localDateMonth, localDateYear);
47 double sunMeanAnomalyRad = SunMeanAnomaly(
48 lctHour, lctMin, lctSec, daylightSaving, zoneCorrectionHours,
49 localDateDay, localDateMonth, localDateYear);
50 double lmDeg = UnwindDeg(13.1763966 * dDays + l0);
51 double mmDeg = UnwindDeg(lmDeg - 0.1114041 * dDays - p0);
52 double nDeg = UnwindDeg(n0 - (0.0529539 * dDays));
53 double evDeg =
54 1.2739 * sin(DegreesToRadians(2.0 * (lmDeg - sunLongDeg) - mmDeg));
55 double aeDeg = 0.1858 * sin(sunMeanAnomalyRad);
56 double a3Deg = 0.37 * sin(sunMeanAnomalyRad);
57 double mmdDeg = mmDeg + evDeg - aeDeg - a3Deg;
58 double ecDeg = 6.2886 * sin(DegreesToRadians(mmdDeg));
59 double a4Deg = 0.214 * sin(2.0 * DegreesToRadians(mmdDeg));
60 double ldDeg = lmDeg + evDeg + ecDeg - aeDeg + a4Deg;
61 double vDeg = 0.6583 * sin(2.0 * DegreesToRadians(ldDeg - sunLongDeg));
62 double lddDeg = ldDeg + vDeg;
63 double ndDeg = nDeg - 0.16 * sin(sunMeanAnomalyRad);
64 double y = sin(DegreesToRadians(lddDeg - ndDeg)) * cos(DegreesToRadians(i));
65 double x = cos(DegreesToRadians(lddDeg - ndDeg));
66
67 double moonLongDeg = UnwindDeg(WToDegrees(atan2(y, x)) + ndDeg);
68 double moonLatDeg = WToDegrees(
69 asin(sin(DegreesToRadians(lddDeg - ndDeg)) * sin(DegreesToRadians(i))));
70 double moonRAHours1 = DecimalDegreesToDegreeHours(EclipticRightAscension(
71 moonLongDeg, 0, 0, moonLatDeg, 0, 0, gdateDay, gdateMonth, gdateYear));
72 double moonDecDeg1 = EclipticDeclination(moonLongDeg, 0, 0, moonLatDeg, 0, 0,
73 gdateDay, gdateMonth, gdateYear);
74
75 int moonRAHour = DecimalHoursHour(moonRAHours1);
76 int moonRAMin = DecimalHoursMinute(moonRAHours1);
77 double moonRASec = DecimalHoursSecond(moonRAHours1);
78 double moonDecDeg = DecimalDegreesDegrees(moonDecDeg1);
79 double moonDecMin = DecimalDegreesMinutes(moonDecDeg1);
80 double moonDecSec = DecimalDegreesSeconds(moonDecDeg1);
81
82 return CMoonApproximatePosition(moonRAHour, moonRAMin, moonRASec, moonDecDeg,
83 moonDecMin, moonDecSec);

181 double y = cos(sunDecRad) * sin(sunRARad - moonRARad);
