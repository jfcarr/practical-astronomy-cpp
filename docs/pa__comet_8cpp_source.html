33 pa_data::CometDataElliptical cometInfo =

35
36 double timeSinceEpochYears =
37 (CivilDateToJulianDate(greenwichDateDay, greenwichDateMonth,
38 greenwichDateYear) -
39 CivilDateToJulianDate(0.0, 1, greenwichDateYear)) /
40 365.242191 +
41 greenwichDateYear - cometInfo.epoch_EpochOfPerihelion;
42 double mcDeg = 360 * timeSinceEpochYears / cometInfo.period_PeriodOfOrbit;
43 double mcRad = DegreesToRadians(mcDeg - 360 * floor(mcDeg / 360));
44 double eccentricity = cometInfo.ecc_EccentricityOfOrbit;
45 double trueAnomalyDeg = WToDegrees(TrueAnomaly(mcRad, eccentricity));
46 double lcDeg = trueAnomalyDeg + cometInfo.peri_LongitudeOfPerihelion;
47 double rAU = cometInfo.axis_SemiMajorAxisOfOrbit *
48 (1 - eccentricity * eccentricity) /
49 (1 + eccentricity * cos(DegreesToRadians(trueAnomalyDeg)));
50 double lcNodeRad =
51 DegreesToRadians(lcDeg - cometInfo.node_LongitudeOfAscendingNode);
52 double psiRad =
53 asin(sin(lcNodeRad) *
54 sin(DegreesToRadians(cometInfo.incl_InclinationOfOrbit)));
55
56 double y =
57 sin(lcNodeRad) * cos(DegreesToRadians(cometInfo.incl_InclinationOfOrbit));
58 double x = cos(lcNodeRad);
59
60 double ldDeg =
61 WToDegrees(atan2(y, x)) + cometInfo.node_LongitudeOfAscendingNode;
62 double rdAU = rAU * cos(psiRad);
63
64 double earthLongitudeLeDeg =
65 SunLong(lctHour, lctMin, lctSec, daylightSaving, zoneCorrectionHours,
66 localDateDay, localDateMonth, localDateYear) +
67 180.0;
68 double earthRadiusVectorAU =
69 SunDist(lctHour, lctMin, lctSec, daylightSaving, zoneCorrectionHours,
70 localDateDay, localDateMonth, localDateYear);
71
72 double leLdRad = DegreesToRadians(earthLongitudeLeDeg - ldDeg);
73 double aRad = (rdAU < earthRadiusVectorAU)
74 ? atan2((rdAU * sin(leLdRad)),
75 (earthRadiusVectorAU - rdAU * cos(leLdRad)))
76 : atan2((earthRadiusVectorAU * sin(-leLdRad)),
77 (rdAU - earthRadiusVectorAU * cos(leLdRad)));
78
79 double cometLongDeg1 = (rdAU < earthRadiusVectorAU)
80 ? 180.0 + earthLongitudeLeDeg + WToDegrees(aRad)
81 : WToDegrees(aRad) + ldDeg;
82 double cometLongDeg = cometLongDeg1 - 360 * floor(cometLongDeg1 / 360);
83 double cometLatDeg = WToDegrees(
84 atan(rdAU * tan(psiRad) * sin(DegreesToRadians((cometLongDeg1 - ldDeg))) /
85 (earthRadiusVectorAU * sin(-leLdRad))));
86 double cometRAHours1 = DecimalDegreesToDegreeHours(EclipticRightAscension(
87 cometLongDeg, 0, 0, cometLatDeg, 0, 0, greenwichDateDay,
88 greenwichDateMonth, greenwichDateYear));
89 double cometDecDeg1 = EclipticDeclination(
90 cometLongDeg, 0, 0, cometLatDeg, 0, 0, greenwichDateDay,
91 greenwichDateMonth, greenwichDateYear);
92 double cometDistanceAU = sqrt(
93 pow(earthRadiusVectorAU, 2) + pow(rAU, 2) -
94 2.0 * earthRadiusVectorAU * rAU *
95 cos(DegreesToRadians((lcDeg - earthLongitudeLeDeg))) * cos(psiRad));
96
97 int cometRAHour = DecimalHoursHour(cometRAHours1 + 0.008333);
98 int cometRAMin = DecimalHoursMinute(cometRAHours1 + 0.008333);
99 double cometDecDeg = DecimalDegreesDegrees(cometDecDeg1 + 0.008333);
100 double cometDecMin = DecimalDegreesMinutes(cometDecDeg1 + 0.008333);
101 double cometDistEarth = Round(cometDistanceAU, 2);
102
103 return CCometPosition(cometRAHour, cometRAMin, 0, cometDecDeg, cometDecMin, 0,
104 cometDistEarth);
105}
