20double PACoordinates::AngleToDecimalDegrees(double degrees, double minutes,
21 double seconds) {
22 double a = std::abs(seconds) / 60;
23 double b = (std::abs(minutes) + a) / 60;
24 double c = std::abs(degrees) + b;
25 double d = (degrees < 0 || minutes < 0 || seconds < 0) ? -c : c;
26
27 return d;
28}

37std::tuple<double, double, double>
38PACoordinates::DecimalDegreesToAngle(double decimalDegrees) {
39 double unsignedDecimal = std::abs(decimalDegrees);
40 double totalSeconds = unsignedDecimal * 3600;
41 double seconds2dp = Round(std::fmod(totalSeconds, 60), 2);
42 double correctedSeconds = (seconds2dp == 60) ? 0 : seconds2dp;
43 double correctedRemainder =
44 (seconds2dp == 60) ? totalSeconds + 60 : totalSeconds;
45 double minutes = std::fmod(floor(correctedRemainder / 60), 60);
46 double unsignedDegrees = floor(correctedRemainder / 3600);
47 double signedDegrees =
48 (decimalDegrees < 0) ? -1 * unsignedDegrees : unsignedDegrees;
49
50 return std::tuple<double, double, double>{signedDegrees, minutes,
51 floor(correctedSeconds)};
52}

492 int epoch1Year, double epoch2Day,

534PACoordinates::NutationInEclipticLongitudeAndObliquity(double greenwichDay,
535 int greenwichMonth,
536 int greenwichYear) {

538 CivilDateToJulianDate(greenwichDay, greenwichMonth, greenwichYear);
539 double tCenturies = (jdDays - 2415020) / 36525;
540 double aDeg = 100.0021358 * tCenturies;
541 double l1Deg = 279.6967 + (0.000303 * tCenturies * tCenturies);
542 double lDeg1 = l1Deg + 360 * (aDeg - floor(aDeg));
543 double lDeg2 = lDeg1 - 360 * floor(lDeg1 / 360);

548 double nRad = DegreesToRadians(nDeg2);
549 double nutInLongArcsec = -17.2 * sin(nRad) - 1.3 * sin(2 * lRad);
550 double nutInOblArcsec = 9.2 * cos(nRad) + 0.5 * cos(2 * lRad);
551
552 double nutInLongDeg = nutInLongArcsec / 3600;
553 double nutInOblDeg = nutInOblArcsec / 3600;
554
555 return std::tuple<double, double>{nutInLongDeg, nutInOblDeg};

696std::tuple<double, double> PACoordinates::HeliographicCoordinates(
697 double helioPositionAngleDeg, double helioDisplacementArcmin,
698 double gwdateDay, int gwdateMonth, int gwdate_year) {
699 double julianDateDays =
700 CivilDateToJulianDate(gwdateDay, gwdateMonth, gwdate_year);
701 double tCenturies = (julianDateDays - 2415020) / 36525;
702 double longAscNodeDeg =
703 DegreesMinutesSecondsToDecimalDegrees(74, 22, 0) + (84 * tCenturies / 60);
704 double sunLongDeg =
705 SunLong(0, 0, 0, 0, 0, gwdateDay, gwdateMonth, gwdate_year);
706 double y =
707 sin(DegreesToRadians(longAscNodeDeg - sunLongDeg)) *
708 cos(DegreesToRadians(DegreesMinutesSecondsToDecimalDegrees(7, 15, 0)));
709 double x = -cos(DegreesToRadians(longAscNodeDeg - sunLongDeg));
710 double aDeg = WToDegrees(atan2(y, x));
711 double mDeg1 = 360 - (360 * (julianDateDays - 2398220) / 25.38);
712 double mDeg2 = mDeg1 - 360 * floor(mDeg1 / 360);
713 double l0Deg1 = mDeg2 + aDeg;
714 double b0Rad = asin(
715 sin(DegreesToRadians(sunLongDeg - longAscNodeDeg)) *
716 sin(DegreesToRadians(DegreesMinutesSecondsToDecimalDegrees(7, 15, 0))));
717 double theta1Rad = atan(-cos(DegreesToRadians(sunLongDeg)) *
718 tan(DegreesToRadians(pa_macros::Obliq(
719 gwdateDay, gwdateMonth, gwdate_year))));
720 double theta2Rad = atan(
721 -cos(DegreesToRadians(longAscNodeDeg - sunLongDeg)) *
722 tan(DegreesToRadians(DegreesMinutesSecondsToDecimalDegrees(7, 15, 0))));
723 double pDeg = WToDegrees(theta1Rad + theta2Rad);
724 double rho1Deg = helioDisplacementArcmin / 60;
725 double rhoRad =
726 asin(2 * rho1Deg /
727 SunDia(0, 0, 0, 0, 0, gwdateDay, gwdateMonth, gwdate_year)) -
728 DegreesToRadians(rho1Deg);
729 double bRad = asin(sin(b0Rad) * cos(rhoRad) +
730 cos(b0Rad) * sin(rhoRad) *
731 cos(DegreesToRadians(pDeg - helioPositionAngleDeg)));
732 double bDeg = WToDegrees(bRad);
733 double lDeg1 =
734 WToDegrees(asin(sin(rhoRad) *
735 sin(DegreesToRadians(pDeg - helioPositionAngleDeg)) /
736 cos(bRad))) +
737 l0Deg1;
738 double lDeg2 = lDeg1 - 360 * floor(lDeg1 / 360);
739
740 double helioLongDeg = Round(lDeg2, 2);
741 double helioLatDeg = Round(bDeg, 2);
742
743 return std::tuple<double, double>{helioLongDeg, helioLatDeg};
744}

767std::tuple<double, double, double>
768PACoordinates::SelenographicCoordinates1(double gwdateDay, int gwdateMonth,
769 int gwdateYear) {
770 double julianDateDays =
771 CivilDateToJulianDate(gwdateDay, gwdateMonth, gwdateYear);
772 double tCenturies = (julianDateDays - 2451545) / 36525;
773 double longAscNodeDeg = 125.044522 - 1934.136261 * tCenturies;
774 double f1 = 93.27191 + 483202.0175 * tCenturies;
775 double f2 = f1 - 360 * floor(f1 / 360);
776 double geocentricMoonLongDeg = pa_macros::MoonLongitude(
777 0, 0, 0, 0, 0, gwdateDay, gwdateMonth, gwdateYear);
778 double geocentricMoonLatRad = DegreesToRadians(pa_macros::MoonLatitude(
779 0, 0, 0, 0, 0, gwdateDay, gwdateMonth, gwdateYear));
780 double inclinationRad =
781 DegreesToRadians(DegreesMinutesSecondsToDecimalDegrees(1, 32, 32.7));
782 double nodeLongRad = DegreesToRadians(longAscNodeDeg - geocentricMoonLongDeg);
783 double sinBe =
784 -cos(inclinationRad) * sin(geocentricMoonLatRad) +
785 sin(inclinationRad) * cos(geocentricMoonLatRad) * sin(nodeLongRad);
786 double subEarthLatDeg = WToDegrees(asin(sinBe));
787 double aRad = atan2(-sin(geocentricMoonLatRad) * sin(inclinationRad) -
788 cos(geocentricMoonLatRad) * cos(inclinationRad) *
789 sin(nodeLongRad),
790 cos(geocentricMoonLatRad) * cos(nodeLongRad));
791 double aDeg = WToDegrees(aRad);
792 double subEarthLongDeg1 = aDeg - f2;
793 double subEarthLongDeg2 =
794 subEarthLongDeg1 - 360 * floor(subEarthLongDeg1 / 360);
795 double subEarthLongDeg3 =
796 (subEarthLongDeg2 > 180) ? (subEarthLongDeg2 - 360) : subEarthLongDeg2;
797 double c1Rad = atan(
798 cos(nodeLongRad) * sin(inclinationRad) /
799 (cos(geocentricMoonLatRad) * cos(inclinationRad) +
800 sin(geocentricMoonLatRad) * sin(inclinationRad) * sin(nodeLongRad)));
801 double obliquityRad =
802 DegreesToRadians(Obliq(gwdateDay, gwdateMonth, gwdateYear));
803 double c2Rad =
804 atan(sin(obliquityRad) * cos(DegreesToRadians(geocentricMoonLongDeg)) /
805 (sin(obliquityRad) * sin(geocentricMoonLatRad) *
806 sin(DegreesToRadians(geocentricMoonLongDeg)) -
807 cos(obliquityRad) * cos(geocentricMoonLatRad)));
808 double cDeg = WToDegrees(c1Rad + c2Rad);
809
810 double subEarthLongitude = Round(subEarthLongDeg3, 2);
811 double subEarthLatitude = Round(subEarthLatDeg, 2);
812 double positionAngleOfPole = Round(cDeg, 2);
813
814 return std::tuple<double, double, double>{subEarthLongitude, subEarthLatitude,
815 positionAngleOfPole};
816}
