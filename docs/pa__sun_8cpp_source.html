24 double greenwichDateDay = LocalCivilTimeGreenwichDay(
25 lctHours, lctMinutes, lctSeconds, daylightSaving, zoneCorrection,
26 localDay, localMonth, localYear);
27 int greenwichDateMonth = LocalCivilTimeGreenwichMonth(
28 lctHours, lctMinutes, lctSeconds, daylightSaving, zoneCorrection,
29 localDay, localMonth, localYear);
30 int greenwichDateYear = LocalCivilTimeGreenwichYear(
31 lctHours, lctMinutes, lctSeconds, daylightSaving, zoneCorrection,
32 localDay, localMonth, localYear);
33 double utHours = LocalCivilTimeToUniversalTime(
34 lctHours, lctMinutes, lctSeconds, daylightSaving, zoneCorrection,
35 localDay, localMonth, localYear);
36 double utDays = utHours / 24;
37 double jdDays = CivilDateToJulianDate(greenwichDateDay, greenwichDateMonth,
38 greenwichDateYear) +
39 utDays;
40 double dDays = jdDays - CivilDateToJulianDate(0, 1, 2010);
41 double nDeg = 360 * dDays / 365.242191;
42 double mDeg1 = nDeg + SunELong(0, 1, 2010) - SunPerigee(0, 1, 2010);
43 double mDeg2 = mDeg1 - 360 * floor(mDeg1 / 360);
44 double eCDeg =
45 360 * SunEccentricity(0, 1, 2010) * sin(DegreesToRadians(mDeg2)) / M_PI;
46 double lSDeg1 = nDeg + eCDeg + SunELong(0, 1, 2010);
47 double lSDeg2 = lSDeg1 - 360 * floor(lSDeg1 / 360);
48 double raDeg =
49 pa_macros::EclipticRightAscension(lSDeg2, 0, 0, 0, 0, 0, greenwichDateDay,
50 greenwichDateMonth, greenwichDateYear);
51 double raHours = DecimalDegreesToDegreeHours(raDeg);
52 double decDeg =
53 pa_macros::EclipticDeclination(lSDeg2, 0, 0, 0, 0, 0, greenwichDateDay,
54 greenwichDateMonth, greenwichDateYear);
55
56 int sunRaHour = DecimalHoursHour(raHours);
57 int sunRaMin = DecimalHoursMinute(raHours);
58 double sunRaSec = DecimalHoursSecond(raHours);
59 double sunDecDeg = DecimalDegreesDegrees(decDeg);
60 double sunDecMin = DecimalDegreesMinutes(decDeg);
61 double sunDecSec = DecimalDegreesSeconds(decDeg);
62
63 return std::tuple<double, double, double, double, double, double>{
64 sunRaHour, sunRaMin, sunRaSec, sunDecDeg, sunDecMin, sunDecSec};

112std::tuple<double, double, double, double> PASun::SunDistanceAndAngularSize(
113 double lctHours, double lctMinutes, double lctSeconds, double localDay,
114 int localMonth, int localYear, bool isDaylightSaving, int zoneCorrection) {
115 int daylightSaving = (isDaylightSaving) ? 1 : 0;
116
117 double gDay = LocalCivilTimeGreenwichDay(lctHours, lctMinutes, lctSeconds,
118 daylightSaving, zoneCorrection,
119 localDay, localMonth, localYear);
120 int gMonth = LocalCivilTimeGreenwichMonth(lctHours, lctMinutes, lctSeconds,
121 daylightSaving, zoneCorrection,
122 localDay, localMonth, localYear);
123 int gYear = LocalCivilTimeGreenwichYear(lctHours, lctMinutes, lctSeconds,
124 daylightSaving, zoneCorrection,
125 localDay, localMonth, localYear);
126 double trueAnomalyDeg =
127 SunTrueAnomaly(lctHours, lctMinutes, lctSeconds, daylightSaving,
128 zoneCorrection, localDay, localMonth, localYear);
129 double trueAnomalyRad = DegreesToRadians(trueAnomalyDeg);
130 double eccentricity = SunEccentricity(gDay, gMonth, gYear);
131 double f = (1 + eccentricity * cos(trueAnomalyRad)) /
132 (1 - eccentricity * eccentricity);
133 double rKm = 149598500 / f;
134 double thetaDeg = f * 0.533128;
135
136 double sunDistKm = Round(rKm, 0);
137 double sunAngSizeDeg = DecimalDegreesDegrees(thetaDeg);
138 double sunAngSizeMin = DecimalDegreesMinutes(thetaDeg);
139 double sunAngSizeSec = DecimalDegreesSeconds(thetaDeg);
140
141 return std::tuple<double, double, double, double>{
142 sunDistKm, sunAngSizeDeg, sunAngSizeMin, sunAngSizeSec};
